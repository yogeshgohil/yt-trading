# Git Commands for Deployment

# Step 1: Initialize Git (run once)
git init

# Step 2: Add all files
git add .

# Step 3: Commit
git commit -m "Initial commit - Trading App"

# Step 4: Add your GitHub repository
git remote add origin https://github.com/YOUR_USERNAME/trading-app.git

# Step 5: Push to GitHub
git branch -M main
git push -u origin main
